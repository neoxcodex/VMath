228 static Mat4 genExtrinsic(dim3::Vec3 cameraPos, dim3::Vec3 targetPos){
229 dim3::Vec3 forward = dim3::normalize(targetPos - cameraPos);
230
231 dim3::Vec3 right = dim3::normalize(dim3::crossProduct({0.0f, 1.0f, 0.0f}, forward));
232 dim3::Vec3 up = dim3::crossProduct(forward, right);
233 Mat4 res = Mat4::identity();
234
235 res.M[0][0] = right.x;
236 res.M[0][1] = right.y;
237 res.M[0][2] = right.z;
238 res.M[1][0] = up.x;
239 res.M[1][1] = up.y;
240 res.M[1][2] = up.z;
241
242 res.M[2][0] = forward.x;
243 res.M[2][1] = forward.y;
244 res.M[2][2] = forward.z;
245
246 res.M[0][3] = -(dim3::dotProduct(right, cameraPos));
247 res.M[1][3] = -(dim3::dotProduct(up, cameraPos));
248 res.M[2][3] = -(dim3::dotProduct(forward, cameraPos));
249 return res;
